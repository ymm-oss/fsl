1spec ShoppingCart {
2 type UserId = 0..1
3 type ItemId = 0..1
4 type Qty = 0..3
5
6 state {
7 stock: Map<ItemId, Qty>,
8 cart: Map<UserId, Option<ItemId>>
9 }
10
11 init {
12 forall i: ItemId { stock[i] = 1 }
13 forall u: UserId { cart[u] = none }
14 }
15
16 action add_to_cart(u: UserId, i: ItemId) {
17 requires cart[u] == none
18 cart[u] = some(i)
19 }
20
21 action remove_from_cart(u: UserId) {
22 requires cart[u] != none
23 cart[u] = none
24 }
25
26 action checkout(u: UserId) {
27 requires cart[u] is some(i)
28 requires stock[i] > 0
29 stock[i] = stock[i] - 1
30 cart[u] = none
31 ensures stock[i] == old(stock[i]) - 1

36 }
37}
